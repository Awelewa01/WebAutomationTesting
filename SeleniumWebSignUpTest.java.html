import io.github.bonigarcia.wdm.WebDriverManager;
import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.chrome.ChromeOptions;
import org.testng.annotations.AfterTest;
import org.testng.annotations.BeforeTest;
import org.testng.annotations.Test;

import static org.openqa.selenium.By.id;

public class SeleniumWebSignUpTest {
    //import the Selenium WebDriver
    private WebDriver driver;

    @BeforeTest

    //creating a new method for signup before the flow
    //setup can be any name
    public void setup() throws InterruptedException {
        //locate where the chrome driver is in the source code
        System.setProperty("webdriver.chrome.driver", "resources/chromedriver.exe");

        //1. open your chrome browser
        ChromeOptions option = new ChromeOptions();
        option.addArguments("--remote-allow-origins=*");
        WebDriverManager.chromedriver().setup();

        driver = new ChromeDriver(option);  //initiates driver
        //2. input your Selenium Demo page URL
        driver.get("https://selenium-blog.herokuapp.com");
        //3. maximize page
        driver.manage().window().maximize();

            //Test 1: verify that user inputs the right url and he is on the right web page
            if (driver.getCurrentUrl().contains("https://selenium-blog.herokuapp.com"))
                System.out.println("Correct Web Page");
            else
                System.out.println("Wrong Web Page");

        //4. locate and click on signup button
        driver.findElement(By.xpath("/html/body/div[2]/div/a[2]")).click();
        Thread.sleep(5000);
        }

    @Test (priority=0)
    public void signupurl() {

        //Test2: verify that user is on the right sign up page
        if(driver.getCurrentUrl().contains("https://selenium-blog.herokuapp.com/signup"))
            System.out.println("Correct Sign up page");
        else
            System.out.println("Wrong Sign up page");
    }

    @Test (priority = 1)
    public void positiveSignup() throws InterruptedException {

        //5. input username on the username field
        driver.findElement(id("user_username")).sendKeys("laptop13");
        //6. input email on the email field
        driver.findElement(id("user_email")).sendKeys("laptop13@mailinator.com");
        //7. input password on the password field
        driver.findElement(id("user_password")).sendKeys("admin");
        //8. click on signup button
        driver.findElement(id("submit")).click();

        //Test3: verify that user can sign up when valid details are inputted
        if(driver.getCurrentUrl().contains("https://selenium-blog.herokuapp.com/users"))
            System.out.println("Valid Sign up Details inputted");
        else
            System.out.println("Incorrect details");

        Thread.sleep(5000);
    }
    @Test (priority = 2)
    public void clickuser1item() throws InterruptedException {

        //9. click on the user1 item on the listing page
        driver.findElement(By.xpath("/html/body/div[2]/div[1]/ul/div/div/li[1]/a")).click();
        Thread.sleep(5000);
    }
    @Test (priority = 3)
    public void verifyitem() throws InterruptedException {

        //10. search for an item (Using Python with Selenium) and confirm if it is present
        driver.findElement(By.xpath("/html/body/div[2]/div[1]/div/div/div[1]/a")).click();

        //Test 4: verify the title page
        String actualpagetitle = "AlphaBlog";
        String expectedpagetitle = driver.getTitle();
        if (actualpagetitle.contains(expectedpagetitle))
            System.out.println("Correct Page Title");
        else
            System.out.println("Wrong Page Title");
        Thread.sleep(5000);
    }
    @Test (priority = 4)
    public void logoutsuccesfull(){
    //11. click on logout
        driver.findElement(By.xpath("//*[@id=\"bs-example-navbar-collapse-1\"]/ul/li[3]/a")).click();
        //Test 5: verify that when user logs out, he is directed to the home page
        String expectedlogouturl="https://selenium-blog.herokuapp.com/";
        String actuallogouturl=driver.getCurrentUrl();
        if(actuallogouturl.contains(expectedlogouturl))
            System.out.println("Log out Successful");
        else
            System.out.println("Logout Not Successful");

    }

    @Test (priority = 5)
    public void negativeSignup() throws InterruptedException {

        //2. input your Selenium Demo page URL
        driver.get("https://selenium-blog.herokuapp.com");
        //3. maximize page
        driver.manage().window().maximize();
        //4. locate and click on signup button
        driver.findElement(By.xpath("/html/body/div[2]/div/a[2]")).click();

        //5. input username on the username field
        driver.findElement(By.id("user_username")).sendKeys("se");
        //6. input email on the email field
        driver.findElement(By.id("user_email")).sendKeys("se@mailinator.com");
        //7. input password on the password field
        driver.findElement(By.id("user_password")).sendKeys("admin");
        //8. click on signup button
        driver.findElement(By.id("submit")).click();

        //Test 6: verify that user cannot login with characters less than two
        if(driver.getCurrentUrl().contains("https://selenium-blog.herokuapp.com/users."))
            System.out.println("Valid Username");
        else
            System.out.println("Invalid Username");
        Thread.sleep(5000);
    }
    @AfterTest
    public void closeBrowser(){
        //close the browser
        driver.quit();
    }
}
